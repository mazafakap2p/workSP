Mainprogram Pg33
Var 
    Int16_t Aa11, Bb22, Cc33, Cc44, Xx55;
Start
    Input Aa11;
    Input Bb22;

    For Cc33 :> Aa11 To Bb22 Do
        (Cc33 Mul Cc33) :> Xx55;
        Output Xx55;
    EndFor

    0 :> Xx55;
    For Cc33 :> 1 To Aa11 Do
        For Cc44 :> 1 To Bb22 Do
            (Xx55 + 1) :> Xx55;
        EndFor
    EndFor
    Output Xx55;
End
