Mainprogram Pg33
Var 
    Int16_t aa11, bb22, cc33, cc44, xx55;
Start
    Input aa11;
    Input bb22;

    For cc33 :> aa11 To bb22 Do
        (cc33 Mul cc33) :> xx55;
        Output Xx55;
    EndFor

    0 :> Xx55;
    For cc33 :> 1 To aa11 Do
        For cc44 :> 1 To bb22 Do
            (xx55 + 1) :> xx55;
        EndFor
    EndFor
    Output xx55;
End
