Mainprogram Pg11
Var 
    Int16_t Aa11, Bb22, Xx33, Yy44;
Start
    Input Aa11;
    Input Bb22;

    Output (Aa11 + Bb22);
    Output (Aa11 - Bb22);
    Output (Aa11 Mul Bb22);
    Output (Aa11 Div Bb22);
    Output (Aa11 Mod Bb22);

    (Aa11 - Bb22) Mul 10 + (Aa11 + Bb22) Div 10 :> Xx33;
    Xx33 + (Xx33 Mod 10) :> Yy44;

    Output Xx33;
    Output Yy44;
End
